/*
Operações Matematicas
+ Adição
- Subtração
/ Divisão
* Multiplicação
Ler 2 valores e mostar o resultado com as 4 operações matematicas*/

int num1, num2, soma, subtracao, multiplicacao;
double divisao;
Console.Write("Informe o primeiro valor - ");
num1 = int.Parse(Console.ReadLine());
Console.Write("Informe o segundo valor - ");
num2 = int.Parse(Console.ReadLine());
soma = num1 + num2;
subtracao = num1 - num2;
multiplicacao = num1 * num2;
divisao = num1 / num2 ;
Console.WriteLine("A Soma dos valores é "+soma);
Console.WriteLine("A Subtração dos valores é " + subtracao);
Console.WriteLine("A Multiplicação dos valores é " + multiplicacao);
Console.WriteLine("A Divisão dos valores é " + divisao);
Console.ReadKey();
